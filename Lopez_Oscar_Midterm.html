dict1={}
with open("numbers.txt", "r") as f:
    text = f.read()
    words = text.split()
    table=str.maketrans("","",string.punctuation)
    stripped = [w.translate(table)for w in words]
for line in stripped:
    for num in line:
        if num in dict1:
            dict1[num]=dict1[num]+1
        else:
            dict1[num]=1;
